// Treasure Spawn ----------------------------------------------------------
payg_cas01.gat,1,1,0	script	Treasure_Py01	-1,{
OnRecvCastlePy01:
	end;
OnInit:
	if($boxNumPy01 == 0) end;
	set $@bxPy01, $boxNumPy01;
	callfunc "F_GldTreas","payg_cas01","Py01",$boxNumPy01,$@bxPy01,$@boxIdPy01,1344,286,4,295,13,1;
	end;

OnDied:
	mapannounce "payg_cas01.gat","Foi aberto o Baú do Tesouro",17;
	set $boxNumPy01, $boxNumPy01 -1;
	if($boxNumPy01 == 0) mapannounce "payg_cas01.gat", "Todas os Baú do Tesouro foram aberto.  Você deve esperar outro dia para que o Tesouro volte.",0;
	end;
}

	end;
}
