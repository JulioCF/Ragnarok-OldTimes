		case 1:
			show "[Soldado de Geffen]";
			show "Você deseja deixar as informações no mini-mapa?";
			next;
		switch(selecT("Não Obrigado","Sim Por Favor")){

		}

		show "Por favor escolha a construção para ver.";
		next;
	sM_Menu:
		switch(select("^FF0000Academia de Magia^000000","^00A500Oficina dos Ferreiros^000000","^4444FFTorre de Geffen^000000","Loja de Armas","Loja de Ferramentas","Bar","Estalagem","Limpar todas as indicações do mini-mapa","Cancelar")){
			case 1:
				show "[Soldado de Geffen]";
				show "'A 'Academia de Magia' é a casa dos teóricos mágicos, e o lugar para se tornar Mago.";
				viewpoint 1,61,180,1,0xFF5555;
				break;
			case 2:
				show "[Soldado de Geffen]";
				show "A 'Oficina dos Ferreiros' está ao sudeste da Torre de Geffen.  É onde Mercadores treinam para se tornar Ferreiros.";
				show "Você também pode aprimorar suas armas e equipamentos lá.";
				viewpoint 1,182,59,2,0x44FF44;
				break;
			case 3:
				show "[Soldado de Geffen]";
				show "Os andares inferiores da 'Torre de Geffen' são uma área cheia de monstros.  Pessoas vão treinar lá.";
				show "O andar superior da Torre é a casa da ^0000FF'Guilda dos Bruxos'^000000. Onde Magos podem se tornar Bruxos.";
				viewpoint 1,120,120,3,0x5555FF;
				break;
			case 4:
				show "[Soldado de Geffen]";
				show "Vá para 'Loja de Armas' se equipar com uma boa variedade de armas e equipamentos.";
				viewpoint 1,99,140,4,0xFF00FF;
				break;
			case 5:
				show "[Soldado de Geffen]";
				show "A 'Loja de Equipamentos' é o lugar para ir quando se precisa se reabastecer de itens como poções ou asas de mosca.";
				viewpoint 1,44,86,5,0xFF00FF;
				break;
			case 6:
				show "[Soldado de Geffen]";
				show "O 'Bar' é um ótimo lugar para conhecer pessoas e se divertir.";
				viewpoint 1,138,138,6,0xF0C40F;
				break;
			case 7:
				show "[Soldado de Geffen]";
				show "A 'Estalagem' é um bom lugar para descasar e recuperar seu hp e sp.";
				viewpoint 1,172,174,7,0xFF8000;
				break;
			case 8:
				set @COMPASS_CHECK, 2;
				break;
				
			case 9: goto M_END;
				
		}
			if(@COMPASS_CHECK != 2) next;
			if(@COMPASS_CHECK == 1) goto sM_Menu;

			M_END:
