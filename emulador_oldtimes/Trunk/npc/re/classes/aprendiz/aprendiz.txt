// Funções Gerais do Campo de Treinamento

function	script	F_NewNovGrounds	{

	switch (getarg(0))
	{
		case 0:
			if (countitem(2819) > 0)
				delitem 2819, countitem(2819);
			if (countitem(2820) > 0)
				delitem 2820, countitem(2820);
			if (countitem(2821) > 0)
				delitem 2821, countitem(2821);
			if (countitem(2822) > 0)
				delitem 2822, countitem(2822);
			if (countitem(2823) > 0)
				delitem 2823, countitem(2823);
			if (countitem(2824) > 0)
				delitem 2824, countitem(2824);
			break;
		
		case 1:
			erasequest 7123;
			erasequest 7124;
			erasequest 7126;
			erasequest 7127;
			break;
			
		case 2:
			set nov_1st_cos, 0;
			set nov_2nd_cos, 0;
			set nov_3_swordman, 0;
			set nov_3_archer, 0;
			set nov_3_thief, 0;
			set nov_3_magician, 0;
			set nov_3_acolyte, 0;
			set nov_3_merchant, 0;
			break;
			
		case 3:
			set .@warp1, rand (1, 2);
			if (strcharinfo(3) == "new_1-3")
			{
				if (!getarg(1))
				{
					if (.@warp1 == 1)
						warp "new_2-3", 96, 21;
					else
						warp "new_3-3", 96, 21;
					end;
				}
				else if (getarg(1) == 1)
				{
					if (.@warp1 == 1)
						warp "new_4-3", 96, 21;
					else
						warp "new_5-3", 96, 21;
					end;
				}
			}
			if (strcharinfo(3) == "new_2-3" || strcharinfo(3) == "new_3-3")
			{
				if (!getarg(1))
				{
					warp "new_1-3", 96, 21;
					end;
				}
				else if (getarg(1) == 1)
				{
					if (.@warp1 == 1)
						warp "new_4-3", 96, 21;
					else
						warp "new_5-3", 96, 21;
					end;
				}
			}
			if (strcharinfo(3) == "new_4-3" || strcharinfo(3) == "new_5-3")
			{
				if (!getarg(1))
				{
					warp "new_1-3", 96, 21;
					end;
				}
				else if (getarg(1) == 1)
				{
					if (.@warp1 == 1)
						warp "new_2-3", 96, 21;
					else
						warp "new_3-3", 96, 21;
					end;
				}
			}
			end;
	}
	return;
}
